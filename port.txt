<script>
  function showContent(page) {
    let content = document.getElementById("content");

    if (page === "home") {
      content.innerHTML = `
        <h2>Home</h2>
        <p>Welcome to the homepage! Here, you'll find updates and featured posts.</p>
      `;
    } 
    else if (page === "about") {
      content.innerHTML = `
        <h2>About</h2>
        <p>This page was created using HTML, CSS, and JavaScript.</p>
      `;
    } 
    else if (page === "contact") {
      content.innerHTML = `
        <h2>Contact</h2>
        <p>You can reach us through email at <b>example@email.com</b>.</p>
      `;
    }

    content.classList.remove("fade");
    void content.offsetWidth; // restart animation
    content.classList.add("fade");
  }
</script>
